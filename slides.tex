\section{字体与列表}
\begin{frame}{字体样式}
	这是一个正常字体\\
	\textbf{这是加粗的字体}\\
	{\Large 这是变大的字体}\\
	{\small 这是变小的字体}\\
	\structure{这是structure字体}\\
	\alert{这是alert字体}\\
	\textcolor{green}{这是带颜色的字体}\\
	\href{https://www.baidu.com}{这是个超链接}
\end{frame}

\begin{frame}{列表}
	\structure{无序列表}
	\begin{itemize}
		\item 项目1
		\item 项目2
		\item 项目3
	\end{itemize}
	\structure{有序列表}
	\begin{enumerate}
		\item 项目1
		\item 项目2
		\item 项目3
	\end{enumerate}
	\structure{描述列表}
	\begin{description}
		\item[项目1] 项目1描述
		\item[项目2] 项目2描述  
	\end{description}
\end{frame}
